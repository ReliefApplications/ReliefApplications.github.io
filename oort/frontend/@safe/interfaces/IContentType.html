/**
 * Enum of content types.
 */
export enum ContentType {
    workflow = 'workflow',
    dashboard = 'dashboard',
    form = 'form'
}

/**
 * Content type interface.
 */
export interface IContentType {
    value: string;
    name: string;
    img: string;
    color: string;
    focusColor: string;
}
